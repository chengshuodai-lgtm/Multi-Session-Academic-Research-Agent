1. 下载源码：
   git clone https://gitee.com/yi-xiaoqiang/research_agent.git
2. 编辑 .env
   DEEPSEEK_API_KEY=sk-xxxxxxxxxxxxxxxxxxxxxxxx
3. 进入工作目录
   D:\xxxx\research_agent
4. 启动后端
   conda create -n py314-research-agent python=3.14
   conda activate py314-research-agent
   cd backend
   pip install -r requirements.txt
   uvicorn app.main:app --reload --host 0.0.0.0 --port 8000
5. 启动前端
   cd frontend
   npm install
   npm run dev
6. 浏览器
   http://localhost:3000
